\section{Heat Equation (FTCS Matrix Method, Dirichlet
BCs)}\label{heat-equation-ftcs-matrix-method-dirichlet-bcs}

import numpy as np import matplotlib.pyplot as plt

\chapter{Parameters}\label{parameters}

alpha = 1.0 Nx = 50 T = 0.1 L, R = 0.0, 0.0 \# boundary values

\chapter{Grid and stability}\label{grid-and-stability}

dx = 1.0/(Nx-1) dt\_max = dx\emph{dx/(2}alpha) s = 0.4 dt =
s\emph{dt\_max Nt = int(T/dt) dt = T/Nt r = alpha}dt/dx**2

\chapter{Initial condition}\label{initial-condition}

x = np.linspace(0.0, 1.0, Nx) u0 = np.sin(np.pi * x)

\chapter{Build matrix A and boundary vector
b}\label{build-matrix-a-and-boundary-vector-b}

m = Nx - 2 main = (1 - 2\emph{r) } np.ones(m) off = r * np.ones(m-1) A =
np.diag(main) + np.diag(off,1) + np.diag(off,-1)

b = np.zeros(m) b{[}0{]}, b{[}-1{]} = r\emph{L, r}R

\chapter{Time-stepping}\label{time-stepping}

u = u0.copy() u\_in = u{[}1:-1{]}.copy() snapshots = {[}{]} snap\_every
= max(1, Nt//12)

for j in range(Nt): u\_in = A @ u\_in + b u{[}1:-1{]} = u\_in if j \%
snap\_every == 0 or j == Nt-1: snapshots.append((j*dt, u.copy()))

\chapter{Plot snapshots}\label{plot-snapshots}

plt.figure(figsize=(8,4)) for t\_here, u\_snap in snapshots: plt.plot(x,
u\_snap, label=f''t=\{t\_here:.4f\}``) plt.plot(x, u0, `k--',
label=`initial', linewidth=1.2) plt.xlabel(`x'); plt.ylabel(`u(x,t)')
plt.title(`Heat equation (FTCS matrix form)')
plt.legend(fontsize=`small') plt.grid(True) plt.show()
